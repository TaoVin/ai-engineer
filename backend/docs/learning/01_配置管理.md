# 步骤1：配置管理

## 1.1 知识点概述

配置管理是任何应用程序的基础，它允许我们将应用程序的设置与代码分离，便于在不同环境（开发、测试、生产）中使用不同的配置。

### 核心概念

- **环境变量**：操作系统级别的配置参数
- **配置文件**：存储配置的文件（如 `.env`、`config.yaml`）
- **配置类**：用于加载和验证配置的代码结构
- **依赖注入**：将配置传递给需要它的组件

### 技术栈

- **pydantic-settings**：基于 Pydantic 的配置管理库
- **python-dotenv**：加载 `.env` 文件的库
- **Pydantic**：数据验证库

## 1.6 配置分组与前缀匹配

### 核心问题：配置分组如何匹配前缀？

在 pydantic-settings 中，嵌套模型默认使用**字段名作为前缀**来匹配环境变量。

### 前缀匹配机制

```python
from pydantic import BaseModel
from pydantic_settings import BaseSettings

class DatabaseConfig(BaseModel):
    """数据库配置 - 字段名会自动加上前缀 DATABASE_"""
    HOST: str = "localhost"
    PORT: int = 3306
    USER: str = "root"
    PASSWORD: str = ""
    NAME: str = "test"

class Setting(BaseSettings):
    # 嵌套模型，字段名为 DATABASE
    DATABASE: DatabaseConfig = DatabaseConfig()
```

**匹配规则**：环境变量名 = `嵌套模型字段名` + `_` + `子字段名`

```bash
# .env 文件中
DATABASE_HOST = "192.168.1.100"    # 匹配 -> DATABASE.HOST
DATABASE_PORT = 3306               # 匹配 -> DATABASE.PORT
DATABASE_USER = "admin"            # 匹配 -> DATABASE.USER
DATABASE_PASSWORD = "secret"       # 匹配 -> DATABASE.PASSWORD
DATABASE_NAME = "production"       # 匹配 -> DATABASE.NAME
```

## 1.7 配置校验

### 使用 Pydantic 验证器

```python
from pydantic import BaseModel, Field, field_validator, model_validator
from pydantic_settings import BaseSettings

class DatabaseConfig(BaseModel):
    """数据库配置"""
    HOST: str = "localhost"
    PORT: int = Field(default=3306, ge=1, le=65535)  # 端口范围校验
    USER: str = Field(default="root", min_length=1, max_length=64)
    PASSWORD: str = Field(default="", max_length=128)
    NAME: str = Field(default="test", min_length=1, max_length=64)
    
    @field_validator("PORT")
    @classmethod
    def validate_port(cls, v: int) -> int:
        """验证端口范围"""
        if not 1024 <= v <= 65535:
            raise ValueError("端口必须在 1024-65535 之间")
        return v
    
    @field_validator("HOST")
    @classmethod
    def validate_host(cls, v: str) -> str:
        """验证主机地址"""
        if v and " " in v:
            raise ValueError("主机地址不能包含空格")
        return v.strip()

class Setting(BaseSettings):
    DATABASE: DatabaseConfig = DatabaseConfig()
    SERVER_PORT: int = Field(default=8001, ge=1024, le=65535)
    
    @field_validator("SERVER_PORT")
    @classmethod
    def validate_server_port(cls, v: int) -> int:
        """验证服务器端口"""
        if v in [3306, 5432, 6379]:
            raise ValueError(f"端口 {v} 是数据库/缓存端口，请使用其他端口")
        return v
    
    @model_validator(mode="after")
    def validate_config(self) -> "Setting":
        """模型级别的验证"""
        # 检查数据库和服务器端口不能相同
        if self.DATABASE.PORT == self.SERVER_PORT:
            raise ValueError("数据库端口和服务器端口不能相同")
        return self
```

## 1.8 动态计算配置

### @property 基础用法

`@property` 是 Python 内置装饰器，将方法转换为只读属性访问。

```python
class DatabaseConfig:
    def __init__(self):
        self._host = "localhost"
        self._port = 3306
    
    @property
    def host(self) -> str:
        """getter：通过属性方式访问"""
        return self._host
    
    @host.setter
    def host(self, value: str):
        """setter：通过属性方式设置"""
        if not value:
            raise ValueError("host 不能为空")
        self._host = value
    
    @host.deleter
    def host(self):
        """deleter：通过属性方式删除"""
        del self._host

# 使用
config = DatabaseConfig()
print(config.host)      # 调用 getter，输出: localhost
config.host = "192.168.1.1"  # 调用 setter
# config.host = ""     # 报错：host 不能为空
del config.host         # 调用 deleter
```

### @property 在 Pydantic BaseModel 中的用法

在 Pydantic 模型中，`@property` 的用法略有不同：

```python
from pydantic import BaseModel

class DatabaseConfig(BaseModel):
    HOST: str = "localhost"
    PORT: int = 3306
    USER: str = "root"
    PASSWORD: str = ""
    NAME: str = "test"
    
    @property
    def url(self) -> str:
        """计算属性：数据库连接 URL"""
        return f"mysql+aiomysql://{self.USER}:{self.PASSWORD}@{self.HOST}:{self.PORT}/{self.NAME}"
    
    @property
    def sync_url(self) -> str:
        """计算属性：同步数据库连接 URL"""
        return f"mysql+pymysql://{self.USER}:{self.PASSWORD}@{self.HOST}:{self.PORT}/{self.NAME}"

# 使用
config = DatabaseConfig(HOST="192.168.1.100", PORT=3307)
print(config.url)       # mysql+aiomysql://root:@192.168.1.100:3307/test
print(config.sync_url)  # mysql+pymysql://root:@192.168.1.100:3307/test

# @property 不会包含在序列化结果中
print(config.model_dump())
# {'HOST': '192.168.1.100', 'PORT': 3307, 'USER': 'root', 'PASSWORD': '', 'NAME': 'test'}
# 注意：url 和 sync_url 不在输出中
```

### @property 在 BaseSettings 中的用法


### 嵌套模型中的 @property

```python
from pydantic import BaseModel
from pydantic_settings import BaseSettings

class DatabaseConfig(BaseModel):
    """数据库配置"""
    TYPE: str = "mysql"
    HOST: str = "localhost"
    PORT: int = 3306
    USER: str = "root"
    PASSWORD: str = ""
    NAME: str = "test"
    
    @property
    def url(self) -> str:
        """动态生成数据库连接 URL"""
        if self.TYPE == "mysql":
            return f"mysql+aiomysql://{self.USER}:{self.PASSWORD}@{self.HOST}:{self.PORT}/{self.NAME}"
        elif self.TYPE == "postgres":
            return f"postgresql+asyncpg://{self.USER}:{self.PASSWORD}@{self.HOST}:{self.PORT}/{self.NAME}"
        else:
            raise ValueError(f"不支持的数据库类型: {self.TYPE}")
    
    @property
    def sync_url(self) -> str:
        """动态生成同步数据库连接 URL（用于 Alembic）"""
        if self.TYPE == "mysql":
            return f"mysql+pymysql://{self.USER}:{self.PASSWORD}@{self.HOST}:{self.PORT}/{self.NAME}"
        elif self.TYPE == "postgres":
            return f"postgresql://{self.USER}:{self.PASSWORD}@{self.HOST}:{self.PORT}/{self.NAME}"
        else:
            raise ValueError(f"不支持的数据库类型: {self.TYPE}")
    
    @property
    def async_driver(self) -> str:
        """获取异步驱动名称"""
        drivers = {"mysql": "aiomysql", "postgres": "asyncpg"}
        return drivers.get(self.TYPE, "")

class RedisConfig(BaseModel):
    """Redis 配置"""
    HOST: str = "localhost"
    PORT: int = 6379
    DB: int = 0
    PASSWORD: str = ""
    
    @property
    def url(self) -> str:
        """动态生成 Redis 连接 URL"""
        if self.PASSWORD:
            return f"redis://:{self.PASSWORD}@{self.HOST}:{self.PORT}/{self.DB}"
        return f"redis://{self.HOST}:{self.PORT}/{self.DB}"
    
    @property
    def connection_params(self) -> dict:
        """连接参数"""
        return {
            "host": self.HOST,
            "port": self.PORT,
            "db": self.DB,
            "password": self.PASSWORD or None,
        }

class Setting(BaseSettings):
    DATABASE: DatabaseConfig = DatabaseConfig()
    REDIS: RedisConfig = RedisConfig()
    
    @property
    def is_production(self) -> bool:
        """判断是否为生产环境"""
        return self.ENVIRONMENT == "prod"
    
    @property
    def debug_mode(self) -> bool:
        """调试模式（开发环境自动开启）"""
        return self.DEBUG or not self.is_production

# 使用示例
settings = Setting()

# 访问嵌套模型的 property
print(settings.DATABASE.url)           # mysql+aiomysql://root:@localhost:3306/test
print(settings.DATABASE.sync_url)      # mysql+pymysql://root:@localhost:3306/test
print(settings.DATABASE.async_driver)  # aiomysql
print(settings.REDIS.url)              # redis://localhost:6379/0
print(settings.REDIS.connection_params)  # {'host': 'localhost', 'port': 6379, 'db': 0, 'password': None}

# 访问主配置的 property
print(settings.is_production)  # False
print(settings.debug_mode)     # True
```

### @computed_field（Pydantic v2）

`@computed_field` 是 Pydantic v2 提供的装饰器，与 `@property` 不同，它会将计算结果包含在序列化输出中。

```python
from pydantic import BaseModel, computed_field
from pydantic_settings import BaseSettings

class DatabaseConfig(BaseModel):
    """数据库配置"""
    TYPE: str = "mysql"
    HOST: str = "localhost"
    PORT: int = 3306
    USER: str = "root"
    PASSWORD: str = ""
    NAME: str = "test"
    
    @computed_field
    @property
    def url(self) -> str:
        """计算字段：数据库连接 URL"""
        if self.TYPE == "mysql":
            return f"mysql+aiomysql://{self.USER}:{self.PASSWORD}@{self.HOST}:{self.PORT}/{self.NAME}"
        elif self.TYPE == "postgres":
            return f"postgresql+asyncpg://{self.USER}:{self.PASSWORD}@{self.HOST}:{self.PORT}/{self.NAME}"
        raise ValueError(f"不支持的数据库类型: {self.TYPE}")

class Setting(BaseSettings):
    DATABASE: DatabaseConfig = DatabaseConfig()

# 使用
config = DatabaseConfig(HOST="192.168.1.100")

# @computed_field 会包含在序列化结果中
print(config.model_dump())
# {'TYPE': 'mysql', 'HOST': '192.168.1.100', 'PORT': 3306, 'USER': 'root', 
#  'PASSWORD': '', 'NAME': 'test', 'url': 'mysql+aiomysql://root:@192.168.1.100:3306/test'}
```

### @property 与 @computed_field 对比

| 特性 | @property | @computed_field |
|------|-----------|-----------------|
| 序列化输出 | ❌ 不包含 | ✅ 包含 |
| JSON 序列化 | ❌ 不包含 | ✅ 包含 |
| model_dump() | ❌ 不包含 | ✅ 包含 |
| 使用场景 | 纯计算、辅助方法 | 需要暴露给前端的计算值 |
| 性能 | 每次访问都计算 | 可缓存 |

```python
class Example(BaseModel):
    name: str = "test"
    
    @property
    def upper_name(self) -> str:
        """不包含在序列化中"""
        return self.name.upper()
    
    @computed_field
    @property
    def display_name(self) -> str:
        """包含在序列化中"""
        return f"Name: {self.name}"

example = Example(name="hello")
print(example.upper_name)    # HELLO
print(example.display_name)  # Name: hello

print(example.model_dump())
# {'name': 'hello', 'display_name': 'Name: hello'}
# 注意：upper_name 不在输出中
```

### 使用 validator 进行动态转换

```python
from pydantic import BaseModel, field_validator
from pydantic_settings import BaseSettings

class Setting(BaseSettings):
    DATABASE_URL: str = ""
    DATABASE_HOST: str = "localhost"
    DATABASE_PORT: int = 3306
    DATABASE_USER: str = "root"
    DATABASE_PASSWORD: str = ""
    DATABASE_NAME: str = "test"
    
    @field_validator("DATABASE_URL", mode="before")
    @classmethod
    def build_database_url(cls, v: str, info) -> str:
        """如果 DATABASE_URL 为空，则根据其他字段动态构建"""
        if v:
            return v
        
        # 从 info.data 获取其他字段值
        host = info.data.get("DATABASE_HOST", "localhost")
        port = info.data.get("DATABASE_PORT", 3306)
        user = info.data.get("DATABASE_USER", "root")
        password = info.data.get("DATABASE_PASSWORD", "")
        name = info.data.get("DATABASE_NAME", "test")
        
        return f"mysql+aiomysql://{user}:{password}@{host}:{port}/{name}"
```

```python
from pydantic import SecretStr
from pydantic_settings import BaseSettings

class Setting(BaseSettings):
```

## 1.10 常见问题

### Q1: 环境变量文件不生效

**原因**：
- 文件路径错误
- 环境变量 `ENVIRONMENT` 未设置

```python
# 检查环境变量

```

### Q2: 配置项类型错误

**原因**：
- 环境变量值与定义的类型不匹配
- 缺少必要的类型转换

**解决方案**：
```python
# 确保类型正确
SERVER_PORT: int = 8001  # 环境变量中应为数字，不是字符串
```

- 需要重启应用

```python
# 清除缓存
get_settings.cache_clear()

```
